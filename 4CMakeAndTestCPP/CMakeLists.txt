cmake_minimum_required(VERSION 3.2)

project(4CMakeAndTestCPP VERSION 1.0
                         DESCRIPTION "Learn CMake and use Test Functions"
                         LANGUAGES CXX)

add_library(aexamplelib STATIC
                src/dog.cpp
                src/cat.cpp

                include/dog.h
                include/cat.h)

target_include_directories(aexamplelib PUBLIC include)
target_compile_features(aexamplelib PUBLIC cxx_std_11)

add_executable(executeApp main.cpp) # "executeApp is the final application name generated by cmake
target_link_libraries(executeApp PUBLIC aexamplelib)
